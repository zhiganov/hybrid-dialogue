# node-room Transcript Enrichment Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let a conversation designer import a Zoom call transcript (manual upload or Fireflies) into a node-room, AI-extract anonymized statements, review them, and post them as provenance-marked contributions that are excluded from the harvest by default.

**Architecture:** A new `imports` table plus three `messages` columns (`import_id`, `muted`, `in_harvest`) and a new `author_type='transcript'`. Pure transcript parsing + a Sonnet extraction call produce a preview; the designer curates; commit inserts transcript messages. The thread renders non-muted transcript items interleaved and marked; the harvest/export exclude them unless opted in. Designer-gated throughout via the existing `requireDesigner`.

**Tech Stack:** Next.js 15 App Router, React 19, TypeScript 5.7, `pg`, `@anthropic-ai/sdk`, vitest. Migrations are plain SQL run by `scripts/migrate.ts` (`npm run migrate`).

## Global Constraints

- **IDs are TEXT, not uuid.** `rooms.id`/`participants.id` are TEXT generated by `generateToken()` (nanoid, 21 url-safe chars). `imports.id` follows the same pattern. (Corrects the design spec's "uuid".)
- **`author_type` and `contribution_tag` have CHECK constraints** (`migrations/001_init.sql`). Adding `'transcript'` requires widening the `author_type` CHECK in a migration.
- **Anonymize.** Transcript items never carry a speaker name; `body` is the statement only; `author_type='transcript'`, `participant_id=null`.
- **No em dashes in any copy we author** (UI strings, prompts). Human/AI content is stripped in `textOf` already; keep authored strings em-dash-free.
- **Designer-gated** writes only, via `requireDesigner(req, roomId)` from `@/lib/designer`.
- **Models are generation-tier** (`src/lib/claude.ts` `MODELS`): extraction uses Sonnet (`claude-sonnet-4-6`), never Opus.
- **Manual migrations**: add `migrations/00N_*.sql`; run `npm run migrate`. Vercel/Railway do not auto-migrate.
- **Refinement flagged for review:** mute / include-in-harvest controls live on the manage **Import panel** (not inline in the participant thread), so muted bodies are never sent to participant clients. The thread shows non-muted transcript items interleaved and marked. If inline muting is wanted, it needs the thread endpoint to gate by role first.

```ts
export type Import = {
  id: string;
  roomId: string;
  sourceKind: "upload" | "fireflies";
  sourceLabel: string;
  sourceRef: string | null;
  consentAttested: boolean;
  createdAt: string;
};

// Message type: add
//   importId: string | null; muted: boolean; inHarvest: boolean; sourceLabel: string | null;

// MessageRow: add import_id, muted, in_harvest, and (via join) source_label
// toMessage: map them.

const MESSAGE_SELECT = `
  SELECT m.id, m.room_id, m.author_type, m.participant_id,
         p.display_name AS author_name, m.body, m.contribution_tag, m.created_at,
         m.import_id, m.muted, m.in_harvest, i.source_label
  FROM messages m
  LEFT JOIN participants p ON p.id = m.participant_id
  LEFT JOIN imports i ON i.id = m.import_id
`;

export async function getMessages(roomId: string, sinceId: number): Promise<Message[]> {
  const rows = await query<MessageRow>(
    `${MESSAGE_SELECT} WHERE m.room_id = $1 AND m.id > $2 AND m.muted = false ORDER BY m.id ASC`,
    [roomId, sinceId]
  );
  return rows.map(toMessage);
}

export async function createImport(input: {
  roomId: string; sourceKind: "upload" | "fireflies"; sourceLabel: string; sourceRef?: string | null;
}): Promise<Import> {
  const id = generateToken();
  const rows = await query<ImportRow>(
    `INSERT INTO imports (id, room_id, source_kind, source_label, source_ref, consent_attested)
     VALUES ($1,$2,$3,$4,$5,true) RETURNING *`,
    [id, input.roomId, input.sourceKind, input.sourceLabel, input.sourceRef ?? null]
  );
  return toImport(rows[0]);
}

export async function addTranscriptMessages(
  roomId: string, importId: string, items: { body: string; tag: ContributionTag | null }[]
): Promise<Message[]> {
  const out: Message[] = [];
  for (const it of items) {
    const inserted = await query<{ id: string }>(
      `INSERT INTO messages (room_id, author_type, participant_id, body, contribution_tag, import_id)
       VALUES ($1,'transcript',NULL,$2,$3,$4) RETURNING id`,
      [roomId, it.body, it.tag, importId]
    );
    const rows = await query<MessageRow>(`${MESSAGE_SELECT} WHERE m.id = $1`, [inserted[0].id]);
    out.push(toMessage(rows[0]));
  }
  return out;
}

export async function getHarvestMessages(roomId: string): Promise<Message[]> {
  const rows = await query<MessageRow>(
    `${MESSAGE_SELECT}
     WHERE m.room_id = $1 AND m.muted = false
       AND (m.author_type <> 'transcript' OR m.in_harvest = true)
     ORDER BY m.id ASC`,
    [roomId]
  );
  return rows.map(toMessage);
}

export async function getTranscriptItems(roomId: string): Promise<Message[]> {
  const rows = await query<MessageRow>(
    `${MESSAGE_SELECT} WHERE m.room_id = $1 AND m.author_type = 'transcript' ORDER BY m.id ASC`,
    [roomId]
  );
  return rows.map(toMessage);
}

export async function setMessageFlags(
  roomId: string, messageId: number, flags: { muted?: boolean; inHarvest?: boolean }
): Promise<Message> {
  const sets: string[] = [];
  const vals: unknown[] = [roomId, messageId];
  if (flags.muted !== undefined) { vals.push(flags.muted); sets.push(`muted = $${vals.length}`); }
  if (flags.inHarvest !== undefined) { vals.push(flags.inHarvest); sets.push(`in_harvest = $${vals.length}`); }
  if (sets.length === 0) throw new Error("no flags to set");
  await query(`UPDATE messages SET ${sets.join(", ")} WHERE room_id = $1 AND id = $2`, vals);
  const rows = await query<MessageRow>(`${MESSAGE_SELECT} WHERE m.id = $1`, [messageId]);
  return toMessage(rows[0]);
}
```

```ts
import { CONTRIBUTION_TAGS, type ContributionTag } from "./domain";

export type ExtractedItem = { body: string; tag: ContributionTag | null };

const VTT_NOISE = /^(WEBVTT.*|\d+|\d{2}:\d{2}:\d{2}[.,]\d{3}\s*-->.*)$/;

export function parseTranscript(raw: string): string {
  return raw
    .split(/\r?\n/)
    .filter((line) => !VTT_NOISE.test(line.trim()))
    .join("\n")
    .replace(/\n{3,}/g, "\n\n")
    .trim();
}

export function validateExtracted(value: unknown): ExtractedItem[] {
  if (!Array.isArray(value)) return [];
  const tags = CONTRIBUTION_TAGS as readonly string[];
  const items: ExtractedItem[] = [];
  for (const raw of value) {
    if (!raw || typeof raw !== "object") continue;
    const body = typeof (raw as { body?: unknown }).body === "string" ? (raw as { body: string }).body.trim() : "";
    if (!body) continue;
    const t = (raw as { tag?: unknown }).tag;
    const tag = typeof t === "string" && tags.includes(t) ? (t as ContributionTag) : null;
    items.push({ body, tag });
    if (items.length === 15) break;
  }
  return items;
}
```

```ts
export async function extractContributions(text: string, room: Room): Promise<ExtractedItem[]> {

  const res = await client().messages.create({
    model: MODELS.extract,
```

```ts
const ENDPOINT = "https://api.fireflies.ai/graphql";

export async function fetchFirefliesTranscript(idOrUrl: string): Promise<{ text: string; label: string }> {
  const key = process.env.FIREFLIES_API_KEY;
  if (!key) throw new Error("FIREFLIES_API_KEY is not set");
  const id = idOrUrl.trim().split("/").pop()?.split("?")[0] ?? idOrUrl.trim();
  const res = await fetch(ENDPOINT, {
    method: "POST",
    headers: { "Content-Type": "application/json", Authorization: `Bearer ${key}` },
    body: JSON.stringify({
      query: `query($id: String!){ transcript(id:$id){ title dateString sentences{ text } } }`,
      variables: { id },
    }),
  });
  if (!res.ok) throw new Error(`Fireflies request failed (${res.status})`);
  const json = (await res.json()) as {
    data?: { transcript?: { title?: string; dateString?: string; sentences?: { text: string }[] } };
    errors?: unknown;
  };
  const t = json.data?.transcript;
  if (!t?.sentences?.length) throw new Error("Fireflies transcript not found or empty");
  const text = t.sentences.map((s) => s.text).join("\n");
  const label = t.title ? `${t.title}${t.dateString ? `, ${t.dateString}` : ""}` : "call from Fireflies";
  return { text, label };
}
```

```ts
import type { NextRequest } from "next/server";
import { requireDesigner } from "@/lib/designer";
import { extractContributions } from "@/lib/claude";
import { fetchFirefliesTranscript } from "@/lib/fireflies";

export async function POST(req: NextRequest, ctx: { params: Promise<{ id: string }> }) {
  const { id } = await ctx.params;
  const gate = await requireDesigner(req, id);
  if ("error" in gate) return gate.error;
  const body = ((await req.json().catch(() => null)) ?? {}) as {
    source?: "upload" | "fireflies"; content?: string; firefliesId?: string; sourceLabel?: string; consent?: boolean;
  };
  if (body.consent !== true) return Response.json({ message: "Consent is required to import a call." }, { status: 400 });

  let text = ""; let sourceLabel = ""; let sourceRef: string | null = null;
  try {
    if (body.source === "fireflies") {
      if (!body.firefliesId) return Response.json({ message: "A Fireflies transcript id or URL is required." }, { status: 400 });
      const f = await fetchFirefliesTranscript(body.firefliesId);
      text = f.text; sourceLabel = f.label; sourceRef = body.firefliesId;
    } else {
      if (!body.content?.trim()) return Response.json({ message: "Upload a transcript with some content." }, { status: 400 });
      text = body.content; sourceLabel = body.sourceLabel?.trim() || "call transcript"; sourceRef = body.sourceLabel ?? null;
    }
  } catch (e) {
    return Response.json({ message: e instanceof Error ? e.message : "Could not read the transcript." }, { status: 400 });
  }

  const items = await extractContributions(text, gate.room);
  if (!items.length) return Response.json({ message: "Nothing worth importing was found in that transcript." }, { status: 422 });
  return Response.json({ items, sourceLabel, sourceKind: body.source ?? "upload", sourceRef });
}
```

```ts
export async function PATCH(req: NextRequest, ctx: { params: Promise<{ id: string; messageId: string }> }) {
  const { id, messageId } = await ctx.params;
  const gate = await requireDesigner(req, id);
  if ("error" in gate) return gate.error;

  if (typeof body.muted === "boolean") flags.muted = body.muted;

  if (Object.keys(flags).length === 0) return Response.json({ message: "Nothing to update." }, { status: 400 });
  const message = await setMessageFlags(id, Number(messageId), flags);
  return Response.json({ message });
}
```
